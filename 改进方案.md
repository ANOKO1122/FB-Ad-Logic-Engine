# FB Ad-Intelligence Server 改进方案

## 📋 方案概述

本方案旨在将当前系统从"单机模式"升级为"多用户、服务端主导"的企业级广告决策中枢。核心目标是实现用户隔离、规则持久化、后端自动化执行和可审计的操作日志。

---

## 1. 核心改进：引入"用户上下文 (User Context)"

### 1.1 问题分析

**当前状态：**
- 系统采用"单机模式"思维
- 规则存储在浏览器 `localStorage`，无法多用户共享
- API 调用不携带用户信息，无法实现数据隔离

**目标：**
- 实现用户隔离："谁投的广告谁看，谁设的规则谁管"
- 所有操作必须携带用户身份信息
- 后端根据用户身份过滤数据和权限

```javascript
// src/stores/userStore.js (新建文件)
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'

export const useUserStore = defineStore('user', () => {
  // 状态
  const currentUserId = ref(localStorage.getItem('current_user_id') || '1')
  const users = ref([
    { id: '1', username: '投手A' },
    { id: '2', username: '投手B' },
    { id: '3', username: '管理员' }
  ])

  // 计算属性
  const currentUser = computed(() => {
    return users.value.find(u => u.id === currentUserId.value) || users.value[0]
  })

  // 方法
  function setUserId(userId) {
    currentUserId.value = userId
    localStorage.setItem('current_user_id', userId)
    // 触发自定义事件，通知其他需要刷新的组件
    window.dispatchEvent(new CustomEvent('user-changed', { detail: { userId } }))
  }

  return {
    currentUserId,
    currentUser,
    users,
    setUserId
  }
})
```

### 2.3 账户ID格式统一处理（重要优化）

**问题：** Facebook 的 `account_id` 有两种格式：
- 纯数字：`123456`
- 带前缀：`act_123456`

**解决方案：** 创建统一的格式化函数，确保存储到数据库时格式一致。

```javascript
// server/utils/accountFormatter.js (新建文件)

/**
 * 统一账户ID格式：确保所有账户ID都带有 act_ 前缀
 * @param {string|number} accountId - 原始账户ID（可能是 123456 或 act_123456）
 * @returns {string} - 格式化后的账户ID（act_123456）
 * 
 * ⚠️ 重要：Facebook 账户ID可能非常长，必须保持为字符串类型，避免精度丢失
 */
export function normalizeAccountId(accountId) {
  if (!accountId) return accountId
  
  // ⚠️ 关键：始终转换为字符串，不要使用 Number() 或 parseInt()
  const id = String(accountId).trim()
  
  // 如果已经是 act_ 开头，直接返回
  if (id.startsWith('act_')) {
    return id
  }
  
  // 如果是纯数字，添加 act_ 前缀（但保持字符串格式）
  if (/^\d+$/.test(id)) {
    return `act_${id}` // ⚠️ 这里 id 已经是字符串，拼接后仍然是字符串
  }
  
  // 其他情况直接返回（可能包含特殊字符）
  return id
}

/**
 * 检查账户ID是否属于指定用户
 * @param {string} accountId - 账户ID
 * @param {number} userId - 用户ID
 * @param {object} pool - 数据库连接池
 * @returns {Promise<boolean>}
 */
export async function checkAccountOwnership(accountId, userId, pool) {
  const normalizedId = normalizeAccountId(accountId)
  
  const [rows] = await pool.query(
    'SELECT id FROM ad_accounts WHERE fb_account_id = ? AND owner_id = ? AND is_active = 1',
    [normalizedId, userId]
  )
  
  return rows.length > 0
}
```

## 3. 规则引擎的"重心下沉"（Phase 2 & 3）

### 3.1 问题分析

**当前问题：**
- 规则自动执行依赖前端浏览器保持打开
- 投手关闭浏览器后，监控和自动止损失效
- 无法实现 7x24 小时无人值守

**解决方案：**
- 规则执行改为后端完全自主刷新
- 使用 `node-cron` 定时任务
- 后端独立运行，不依赖前端

### 3.2 实现步骤

**1. 安装依赖**

```bash
npm install node-cron mysql2
```

**2. 创建定时任务服务**

```javascript
// server/services/cronService.js (新建文件)
import cron from 'node-cron'
import pool from '../db/connection.js'
import { FacebookMarketingAPI } from '../index.js'
import { RuleEngine } from '../index.js'

class CronService {
  constructor() {
    this.job = null
  }

  // 启动定时任务（每 15 分钟执行一次）
  start() {
    if (this.job) {
      console.log('⏰ 定时任务已在运行')
      return
    }

    console.log('🚀 启动规则自动执行定时任务（每15分钟）')

    // 使用 cron 表达式：每 15 分钟执行一次
    this.job = cron.schedule('*/15 * * * *', async () => {
      console.log('\n' + '='.repeat(50))
      console.log('⏰ 定时任务触发:', new Date().toLocaleString('zh-CN'))
      await this.executeAllRules()
      console.log('='.repeat(50) + '\n')
    }, {
      scheduled: true,
      timezone: 'Asia/Shanghai'
    })
  }

  // 停止定时任务
  stop() {
    if (this.job) {
      this.job.stop()
      this.job = null
      console.log('⏸️ 定时任务已停止')
    }
  }

  // 执行所有启用的规则
  async executeAllRules() {
    // 并发保护：如果上一个任务还在运行，直接跳过
    if (this.isRunning) {
      console.log('⏭️  上一个任务还在执行中，跳过本次执行')
      return
    }

    this.isRunning = true
    
    try {
      // 1. 从数据库获取所有启用的规则
      const [rules] = await pool.query(`
        SELECT r.*, u.id as user_id
        FROM rules r
        JOIN users u ON r.user_id = u.id
        WHERE r.is_enabled = 1
      `)

      if (rules.length === 0) {
        console.log('📋 当前没有启用的规则')
        return
      }

      console.log(`📋 找到 ${rules.length} 条启用的规则`)

      // 2. 按账户分组，避免重复获取数据
      const accountsMap = new Map()
      for (const rule of rules) {
        const accountId = rule.account_id
        if (!accountsMap.has(accountId)) {
          accountsMap.set(accountId, [])
        }
        accountsMap.get(accountId).push(rule)
      }

      // 3. 为每个账户执行规则
      const token = process.env.FACEBOOK_ACCESS_TOKEN
      if (!token) {
        console.error('❌ FACEBOOK_ACCESS_TOKEN 未配置，跳过执行')
        return
      }

      const api = new FacebookMarketingAPI(token)

      for (const [accountId, accountRules] of accountsMap.entries()) {
        try {
          console.log(`\n📊 处理账户: ${accountId}`)
          
          // 获取账户的当前数据
          const insights = await api.getAdInsights(accountId, { preset: 'today' })
          const ads = await api.getAds(accountId)

          // 执行该账户的所有规则
          const ruleEngine = new RuleEngine(api)
          
          for (const rule of accountRules) {
            console.log(`\n🔍 评估规则: ${rule.rule_name} (用户: ${rule.user_id})`)
            
            // 解析 JSON 字段
            const conditions = JSON.parse(rule.conditions)
            const actions = JSON.parse(rule.actions)
            const ruleConfig = {
              id: rule.id,
              name: rule.rule_name,
              conditions,
              actions,
              enabled: rule.is_enabled,
              dry_run: rule.dry_run === 1
            }

            // 评估规则
            const executed = await ruleEngine.evaluateRule(ruleConfig, insights, ads)

            if (executed) {
              console.log(`✅ 规则 "${rule.rule_name}" 已触发并执行`)
            } else {
              console.log(`⏭️  规则 "${rule.rule_name}" 未触发`)
            }
          }
        } catch (error) {
          console.error(`❌ 处理账户 ${accountId} 时出错:`, error.message)
        }
      }

      console.log('\n✅ 所有规则执行完成')
    } catch (error) {
      console.error('❌ 执行规则任务失败:', error)
    } finally {
      // 确保无论成功还是失败，都释放运行标记
      this.isRunning = false
    }
  }
}

export default new CronService()
```

## 4. 负责人只能看自己账户的逻辑

### 4.1 实现方案

**核心思路：**
1. 前端请求时携带 `x-user-id` 头
2. 后端根据 `user_id` 查询映射表，获取该用户绑定的 `fb_account_id` 列表
3. 调用 Facebook API 获取原始账户列表
4. 过滤：只返回映射表里存在的账户

### 4.2 代码实现

```javascript
// server/index.js (修改 GET /api/accounts 路由)
import pool from './db/connection.js'
import { userAuthMiddleware } from './middleware/auth.js'

app.get('/api/accounts', userAuthMiddleware, async (req, res) => {
  const startTime = Date.now()
  try {
    const userId = req.userId // 从中间件获取
    
    console.log(`\n📥 收到请求: GET /api/accounts (用户ID: ${userId})`)

    const token = process.env.FACEBOOK_ACCESS_TOKEN
    if (!token) {
      return res.status(401).json({ error: 'Facebook访问令牌未配置' })
    }

    // 1. 从数据库查询该用户绑定的账户ID列表
    const [mappedRows] = await pool.query(
      'SELECT fb_account_id FROM ad_accounts WHERE owner_id = ? AND is_active = 1',
      [userId]
    )
    
    const mappedAccountIds = mappedRows.map(row => row.fb_account_id)
    
    if (mappedAccountIds.length === 0) {
      console.log(`⚠️ 用户 ${userId} 未绑定任何账户`)
      return res.json({ accounts: [] })
    }

    console.log(`📋 用户 ${userId} 绑定的账户: ${mappedAccountIds.join(', ')}`)

    // 2. 调用 Facebook API 获取所有账户
    const api = new FacebookMarketingAPI(token)
    const allAccounts = await api.getAdAccounts()

    // 3. 过滤：只返回映射表里存在的账户
    const filteredAccounts = allAccounts.filter(acc => 
      mappedAccountIds.includes(acc.id) || mappedAccountIds.includes(acc.account_id)
    )

    console.log(`✅ 返回 ${filteredAccounts.length} 个账户 (从 ${allAccounts.length} 个中过滤)`)

    res.json({ accounts: filteredAccounts })
  } catch (error) {
    console.error('❌ 获取账户列表失败:', error)
    res.status(500).json({ error: error.message })
  }
})
```

## 5. 增加 "Dry Run（模拟执行）" 模式

### 5.1 实现方案

**目标：**
- 在自动化规则上线初期，让投手先验证规则准确性
- Dry Run 模式下，只记录日志，不实际执行操作
- 增加投手对系统的信任度

### 5.2 后端 RuleEngine 改造

```javascript
// server/index.js (修改 RuleEngine 类)
class RuleEngine {
  constructor(api, pool) {
    this.api = api
    this.pool = pool
  }

  async evaluateRule(rule, insights, ads) {
    if (!rule.enabled) return false

    const isDryRun = rule.dry_run === true

    for (const insight of insights) {
      const ad = ads.find(a => a.id === insight.ad_id)
      if (!ad) continue

      if (this.evaluateConditions(rule.conditions, insight)) {
        console.log(`\n🎯 规则 "${rule.name}" 触发`)
        console.log(`📊 广告: ${ad.name || ad.id}`)
        console.log(`📈 数据快照:`, {
          ctr: insight.ctr,
          cpc: insight.cpc,
          spend: insight.spend,
          conversions: insight.conversions
        })

        if (isDryRun) {
          // 模拟模式：只记录日志，不执行操作
          console.log(`🧪 [模拟执行] 将执行操作:`, rule.actions)
          
          // 写入执行日志
          await this.logExecution(rule, ad, insight, rule.actions, 'SIMULATE', true)
        } else {
          // 实战模式：真实执行操作
          console.log(`⚡ [实战执行] 正在执行操作...`)
          
          try {
            await this.executeActions(rule.actions, ad, insight)
            
            // 写入执行日志
            await this.logExecution(rule, ad, insight, rule.actions, 'REAL', true)
            
            console.log(`✅ 操作执行成功`)
          } catch (error) {
            console.error(`❌ 操作执行失败:`, error.message)
            
            // 记录失败日志
            await this.logExecution(rule, ad, insight, rule.actions, 'REAL', false, error.message)
          }
        }

        return true
      }
    }
    return false
  }

  // 记录执行日志到数据库
  async logExecution(rule, ad, insight, actions, executionMode, success, errorMessage = null) {
    try {
      const snapshotData = {
        ctr: insight.ctr,
        cpc: insight.cpc,
        spend: insight.spend,
        conversions: insight.conversions,
        clicks: insight.clicks,
        impressions: insight.impressions,
        cpm: insight.cpm
      }

      // 获取账户ID（从 insight 或 rule 中）
      const accountId = insight.account_id || rule.account_id

      for (const action of actions) {
        await this.pool.query(`
          INSERT INTO execution_logs 
          (rule_id, user_id, fb_account_id, fb_ad_id, ad_name, rule_name, 
           action_performed, execution_mode, snapshot_data, status, error_message)
          VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        `, [
          rule.id,
          rule.user_id,
          accountId,
          ad.id,
          ad.name || ad.id,
          rule.name,
          action.type,
          executionMode,
          JSON.stringify(snapshotData),
          success ? 'SUCCESS' : 'FAILED',
          errorMessage
        ])
      }
    } catch (error) {
      console.error('❌ 记录执行日志失败:', error)
    }
  }

  // ... 其他方法保持不变
}
```

## 9. 实施注意事项与避坑指南

### 9.1 关键避坑提示（重要）

#### ⚠️ 避坑 1：account_id 的类型处理（String vs BigInt）

**问题：** Facebook 的广告账户 ID 可能非常长（如 `act_123456789012345`），如果中途转换为 Number 类型，会发生精度丢失（溢出）。

**解决方案：** 在整个代码逻辑中，始终保持 `account_id` 为 String 类型。

```javascript
// ❌ 错误做法：转换为数字
const accountId = parseInt('act_123456789012345') // 会丢失精度
const accountId = Number('123456789012345') // 可能溢出

// ✅ 正确做法：保持字符串
const accountId = 'act_123456789012345'
const accountId = String(accountId) // 确保是字符串

// 在数据库查询时也要注意
await pool.query('SELECT * FROM ad_accounts WHERE fb_account_id = ?', [accountId]) // ✅ 字符串传参
```

**检查清单：**
- [ ] 所有涉及 `account_id` 的变量声明都使用 String 类型
- [ ] 避免使用 `parseInt()`、`Number()` 等函数转换账户 ID
- [ ] 数据库查询时确保参数是字符串类型

#### ⚠️ 避坑 2：环境变量的加载优先级

**问题：** 如果 `dotenv.config()` 执行顺序不对，数据库连接池初始化时可能读不到环境变量（如密码）。

**解决方案：** 在 `server/index.js` 的最顶部执行 `dotenv.config()`。

```javascript
// server/index.js
// ✅ 正确：最顶部导入并执行
import dotenv from 'dotenv'
dotenv.config() // 必须在最前面

import express from 'express'
import mysql from 'mysql2/promise'
import pool from './db/connection.js' // 这里会读取环境变量
// ...

// ❌ 错误：在数据库连接之后才执行
import pool from './db/connection.js'
import dotenv from 'dotenv'
dotenv.config() // 太晚了，pool 已经初始化了
```

**检查清单：**
- [ ] `dotenv.config()` 位于 `server/index.js` 第一行
- [ ] 所有其他导入都在 `dotenv.config()` 之后
- [ ] 数据库连接池的创建在 `dotenv.config()` 之后

#### ⚠️ 避坑 3：跨域问题 (CORS) - 自定义请求头

**问题：** 当前端 `localhost:3000` 访问后端 `localhost:3001` 且带有自定义请求头 `x-user-id` 时，浏览器会发起 OPTIONS 预检请求。如果后端没有正确配置 CORS，会导致请求失败。

**解决方案：** 在 Express 的 CORS 中间件中明确配置 `allowedHeaders`。

```javascript
// server/index.js
import express from 'express'
import cors from 'cors'

const app = express()

// ✅ 正确配置：明确允许自定义请求头
app.use(cors({
  origin: 'http://localhost:3000', // 前端地址
  credentials: true, // 如果需要发送 cookies
  allowedHeaders: ['Content-Type', 'Authorization', 'x-user-id'], // 关键：允许自定义请求头
  exposedHeaders: ['x-user-id'], // 可选：允许前端读取的响应头
  methods: ['GET', 'POST', 'PUT', 'DELETE', 'OPTIONS'] // 允许的 HTTP 方法
}))

// ❌ 错误配置：没有明确允许自定义请求头
app.use(cors()) // 可能不支持 x-user-id 头

// 中间件
app.use(express.json())
```

**验证方法：**
打开浏览器开发者工具 → Network 标签，查看请求：
- 如果看到 OPTIONS 请求返回 200，说明 CORS 配置正确
- 如果 OPTIONS 请求失败（403/404），说明 CORS 配置有问题

**检查清单：**
- [ ] CORS 中间件配置了 `allowedHeaders: ['Content-Type', 'x-user-id']`
- [ ] 测试前端请求时，浏览器控制台没有 CORS 错误
- [ ] OPTIONS 预检请求成功返回 200

            <td>{{ mapping.fb_account_id }}</td>
            <td>{{ mapping.fb_account_name }}</td>
            <td>{{ getUserName(mapping.owner_id) }}</td>

                      @change="updateMapping(mapping)">

                <option value="REAL">实战模式</option>

import { ref, onMounted, computed } from 'vue'
import axios from 'axios'

const allAccounts = ref([]) // 从 Facebook API 获取的所有账户
const mappings = ref([]) // 从数据库获取的映射关系
const users = ref([]) // 用户列表

// 计算属性：未分配的账户
const unassignedAccounts = computed(() => {
  const assignedIds = new Set(mappings.value.map(m => m.fb_account_id))
  return allAccounts.value.filter(acc => !assignedIds.has(acc.id))
})

// 计算属性：已分配的账户
const assignedAccounts = computed(() => {
  return mappings.value
})

const loadData = async () => {
  // 加载所有账户
  const accountsRes = await axios.get('/api/accounts-all') // 需要新增一个管理员接口
  allAccounts.value = accountsRes.data.accounts

  // 加载映射关系
  const mappingsRes = await axios.get('/api/account-mappings')
  mappings.value = mappingsRes.data.mappings

  // 加载用户列表
  const usersRes = await axios.get('/api/users')
  users.value = usersRes.data.users
}

const assignAccount = async (account) => {
  if (!account.assignedTo) {
    alert('请选择负责人')
    return
  }

  try {
    await axios.post('/api/account-mappings', {
      fb_account_id: account.id,
      fb_account_name: account.name,
      owner_id: account.assignedTo,
      execution_mode: 'SIMULATE' // 默认模拟模式
    })
    
    alert('分配成功')
    await loadData()
  } catch (error) {
    alert('分配失败: ' + error.message)
  }
}

const removeMapping = async (mappingId) => {
  if (!confirm('确定要取消分配吗？')) return

  try {
    await axios.delete(`/api/account-mappings/${mappingId}`)
    alert('取消分配成功')
    await loadData()
  } catch (error) {
    alert('操作失败: ' + error.message)
  }
}

const updateMapping = async (mapping) => {
  try {
    await axios.put(`/api/account-mappings/${mapping.id}`, {
      execution_mode: mapping.execution_mode
    })
  } catch (error) {
    alert('更新失败: ' + error.message)
  }
}

const getUserName = (userId) => {
  const user = users.value.find(u => u.id === userId)
  return user ? user.username : '未知'
}

onMounted(() => {
  loadData()
})
