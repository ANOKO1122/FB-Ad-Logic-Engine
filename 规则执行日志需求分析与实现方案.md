### 实现3：新增规则运行摘要表

**文件**：`server/db/schema.js`

**新增表结构**：
```javascript
// rule_execution_summary 表（规则运行摘要）
export const ruleExecutionSummary = mysqlTable('rule_execution_summary', {
  // 主键
  id: int('id').primaryKey().autoincrement(),
  
  // 关联信息
  accountId: varchar('account_id', { length: 50 }).notNull(),
  ruleId: int('rule_id').notNull(),
  ruleName: varchar('rule_name', { length: 255 }).notNull(),
  ownerId: int('owner_id').notNull(),
  
  // 执行结果统计
  matchedCount: int('matched_count').default(0),        // 匹配的广告数量
  executedCount: int('executed_count').default(0),      // 成功执行的广告数量
  failedCount: int('failed_count').default(0),          // 执行失败的广告数量
  skippedCount: int('skipped_count').default(0),        // 跳过的广告数量
  
  // 跳过原因（JSON格式）
  skipReasons: json('skip_reasons'),                     // 跳过原因汇总
  // 示例：{ "cooldown": 2, "no_permission": 1, "account_mismatch": 1, "zero_match": 5 }
  
  // 错误信息（如果有）
  errorMessage: text('error_message'),                  // 规则执行错误信息
  
  // 执行耗时
  durationMs: int('duration_ms'),                       // 执行耗时（毫秒）
  
  // 时间戳
  executedAt: timestamp('executed_at').notNull(),       // 执行时间
  createdAt: timestamp('created_at').notNull().defaultNow()
})
```

**需要创建迁移脚本**：
```sql
CREATE TABLE IF NOT EXISTS rule_execution_summary (
  id INT AUTO_INCREMENT PRIMARY KEY,
  account_id VARCHAR(50) NOT NULL,
  rule_id INT NOT NULL,
  rule_name VARCHAR(255) NOT NULL,
  owner_id INT NOT NULL,
  matched_count INT DEFAULT 0,
  executed_count INT DEFAULT 0,
  failed_count INT DEFAULT 0,
  skipped_count INT DEFAULT 0,
  skip_reasons JSON,
  error_message TEXT,
  duration_ms INT,
  executed_at TIMESTAMP NOT NULL,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  INDEX idx_account_executed (account_id, executed_at),
  INDEX idx_rule_executed (rule_id, executed_at),
  INDEX idx_owner_executed (owner_id, executed_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

### 实现4：在规则评估时记录摘要

**文件**：`server/services/cronService.js`  
**位置**：规则执行循环（第242-285行）

**修改方案**：
在规则评估的每个分支都记录摘要信息：
- 匹配成功：记录 matchedCount、executedCount
- 匹配失败：记录 skipReasons（zero_match）
- 账户不匹配：记录 skipReasons（account_mismatch）
- 权限不足：记录 skipReasons（no_permission）
- 冷却期：记录 skipReasons（cooldown）
- 执行错误：记录 errorMessage

**实现代码**（伪代码）：
```javascript
// 在规则执行循环中
const summary = {
  accountId: lockedAccountId,
  ruleId: rule.id,
  ruleName: rule.ruleName,
  ownerId: user.owner_id || 0,
  matchedCount: 0,
  executedCount: 0,
  failedCount: 0,
  skippedCount: 0,
  skipReasons: {},
  errorMessage: null,
  durationMs: 0
}

const startTime = Date.now()

try {
  // 规则评估
  const matchedAds = await ruleEngine.evaluateRule(rule, lockedAccountId)
  
  if (matchedAds.length > 0) {
    summary.matchedCount = matchedAds.length
    // 执行动作...
    summary.executedCount = execStats.successCount
    summary.failedCount = execStats.failCount
  } else {
    summary.skipReasons.zero_match = 1
  }
} catch (error) {
  summary.errorMessage = error.message
} finally {
  summary.durationMs = Date.now() - startTime
  // 写入摘要表（只给 admin 看）
  await insertRuleExecutionSummary(summary)
}
```
