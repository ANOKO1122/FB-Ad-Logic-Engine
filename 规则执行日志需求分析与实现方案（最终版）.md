#### 优化点4：前端时区显示统一（方案外优化）

**背景**：
- 后端存储时间统一为 UTC
- 业务逻辑计算使用账户时区
- 前端展示需要统一时区，避免因用户浏览器时区不同导致显示不一致

**实现方案**：
- 创建前端时区工具模块：`src/utils/dateTime.js`
- 使用 Luxon 库统一将后端返回的 UTC 时间转换为北京时区（UTC+8 / Asia/Shanghai）
- 更新所有时间展示组件（`Logs.vue`、`SystemStatus.vue`）使用统一时区工具

**实现代码**：

**文件**：`src/utils/dateTime.js`
```javascript
import { DateTime } from 'luxon'

/** 前端展示统一使用的时区 */
const DISPLAY_TIMEZONE = 'Asia/Shanghai' // UTC+8 北京时区

/**
 * 解析后端返回的 UTC 时间字符串
 */
export function parseUTC(dateStr) {
  return DateTime.fromISO(dateStr, { zone: 'utc' })
}

/**
 * 将 UTC 时间转为北京时区显示（仅时间 HH:mm:ss）
 */
export function formatTimeBeijing(dateStr) {
  if (!dateStr) return ''
  return parseUTC(dateStr).setZone(DISPLAY_TIMEZONE).toFormat('HH:mm:ss')
}

/**
 * 将 UTC 时间转为北京时区显示（仅日期 MM/dd）
 */
export function formatDateBeijing(dateStr) {
  if (!dateStr) return ''
  return parseUTC(dateStr).setZone(DISPLAY_TIMEZONE).toFormat('MM/dd')
}

/**
 * 将 UTC 时间转为北京时区显示（日期+时间，完整）
 */
export function formatDateTimeBeijing(dateStr) {
  if (!dateStr) return ''
  return parseUTC(dateStr).setZone(DISPLAY_TIMEZONE).toFormat('yyyy-MM-dd HH:mm:ss')
}

/**
 * 将 UTC 时间转为北京时区显示（中文友好格式，用于列表/详情）
 */
export function formatDateTimeBeijingLocale(dateStr) {
  if (!dateStr) return ''
  return parseUTC(dateStr).setZone(DISPLAY_TIMEZONE).toLocaleString(DateTime.DATETIME_MED)
}
```

### 实现2：在规则评估时记录摘要

**文件**：`server/services/cronService.js`

#### 需要覆盖的所有分支

根据代码分析，需要记录摘要的分支包括：

1. **用户不存在或未激活**（`cronService.js:L212-214`）
   - `skip_reason = 'user_not_found'`
   - `skip_details = { "user_id": userId }`

2. **用户无权访问账户**（`cronService.js:L234-236`）
   - `skip_reason = 'no_permission'`
   - `skip_details = { "user_id": userId, "account_id": lockedAccountId }`

3. **规则账户不匹配**（`cronService.js:L247-249`）
   - `skip_reason = 'account_mismatch'`
   - `skip_details = { "rule_account_id": ruleAccountId, "locked_account_id": lockedAccountId }`

4. **规则评估匹配0个广告**（`cronService.js:L275-276`）
   - `skip_reason = 'no_match'`
   - `matched_count = 0`

5. **规则评估异常**（`cronService.js:L281-284`）
   - `skip_reason = 'error'`
   - `error_message = sanitizeErrorMessage(error.message)`

6. **执行动作异常**（`cronService.js:L271-273`）
   - `skip_reason = 'error'`
   - `failed_count > 0`
   - `error_message = sanitizeErrorMessage(execError.message)`

7. **用户处理异常**（`cronService.js:L286-289`）
   - `skip_reason = 'error'`
   - `error_message = sanitizeErrorMessage(error.message)`

**注意**：冷却期和规则禁用的检查在 `getRulesForAccount` 中，在进入用户循环之前。如果需要记录这些情况，可以在 `executeRulesForAccount` 的入口处记录。

#### run_id 的生成与传播

**需要覆盖的 3 个入口**：
1. **cron 触发**：`executeAllRules()` → `executeRulesForAccount()`
2. **手动触发**：`manualExecute()` → `executeAllRules()` → `executeRulesForAccount()`
3. **事件触发**：`ingestorService.js` L2759-2761 直接调用 `executeRulesForAccount()`

**实现方案**：
- 在 `executeAllRules()` 中生成 `run_id`，传递给所有 `executeRulesForAccount()` 调用
- 在事件触发的 `executeRulesForAccount()` 调用中，也生成 `run_id`
- `run_id` 格式：`${Date.now()}-${Math.random().toString(36).substr(2, 9)}` 或 UUID

#### 实现代码结构

**在 `executeRulesForAccount` 函数中**：

```javascript
export async function executeRulesForAccount(accountId, options = {}) {
  const { force = false, runId = null } = options  // ✅ 支持外部传入 run_id
  
  // ✅ 生成 run_id（如果未传入，则生成新的；事件触发时会生成新的）
  const currentRunId = runId || `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`
  
  // ... 现有代码 ...
  
  await executeRulesWithLock(accountId, async (lockedAccountId) => {
    // ... 现有代码 ...
    
    // 为每个用户执行规则
    for (const [userId, userRules] of Object.entries(rulesByUser)) {
      // ... 获取用户信息 ...
      
      // 为每个规则执行
      for (const rule of userRules) {
        const startTime = Date.now()
        const summary = {
          runId: currentRunId,              // ✅ 使用传入或生成的 run_id
          accountId: lockedAccountId,
          ruleId: rule.id,
          ruleName: rule.ruleName,
          userId: userId,                  // ✅ 新增：规则所属用户ID
          ownerId: user.owner_id || 0,
          matchedCount: 0,
          executedCount: 0,
          failedCount: 0,
          skippedCount: 0,
          status: null,                    // ✅ 新增：执行状态（matched/no_match/skipped/error）
          skipReason: null,
          skipDetails: null,               // ✅ 注意：直接存对象，不要 stringify
          errorMessage: null,
          durationMs: 0
        }
        
        try {
          // 分支1：规则账户不匹配
          const ruleAccountId = rule.accountId
          if (ruleAccountId && ruleAccountId !== lockedAccountId) {
            summary.status = 'skipped'
            summary.skipReason = 'account_mismatch'
            summary.skipDetails = {  // ✅ 直接存对象，不要 stringify
              rule_account_id: ruleAccountId,
              locked_account_id: lockedAccountId
            }
            await insertRuleExecutionSummary(summary)
            continue
          }
          
          // 执行规则评估
          const matchedAds = await ruleEngine.evaluateRule(rule, lockedAccountId)
          
          if (Array.isArray(matchedAds) && matchedAds.length > 0) {
            summary.matchedCount = matchedAds.length
            
            // 执行动作
            try {
              const execStats = await executeActionsForRule({
                rule,
                matchedAds,
                accountId: lockedAccountId,
                ownerId: user.owner_id || 0
              })
              
              summary.executedCount = execStats.successCount || 0
              summary.failedCount = execStats.failCount || 0
              summary.status = execStats.failCount > 0 ? 'error' : 'matched'  // ✅ 设置状态
            } catch (execError) {
              summary.status = 'error'
              summary.skipReason = 'error'
              summary.failedCount = matchedAds.length
              summary.errorMessage = sanitizeErrorMessage(execError.message)
            }
          } else {
            // 分支2：规则评估匹配0个广告
            summary.status = 'no_match'
            summary.skipReason = 'no_match'
          }
        } catch (error) {
          // 分支3：规则评估异常
          summary.status = 'error'
          summary.skipReason = 'error'
          summary.errorMessage = sanitizeErrorMessage(error.message)
        } finally {
          summary.durationMs = Date.now() - startTime
          summary.evaluatedAt = new Date()
          await insertRuleExecutionSummary(summary)
        }
      }
    }
  })
}
```

#### evaluated_at 时区转换处理（方案外优化）

**问题**：JavaScript `Date` 对象传递给 MySQL 驱动时，可能因进程时区不同导致时间转换不一致。

**解决方案**：
- 在 `ruleExecutionSummaryService.js` 中，显式将 `evaluatedAt` 转换为 UTC 时间字符串
- 避免依赖进程时区，确保时间存储的一致性

**实现代码**：
```javascript

let evaluatedAt = summary.evaluatedAt || new Date()

  const month = String(evaluatedAt.getUTCMonth() + 1).padStart(2, '0')
  const day = String(evaluatedAt.getUTCDate()).padStart(2, '0')
  const hours = String(evaluatedAt.getUTCHours()).padStart(2, '0')
  const minutes = String(evaluatedAt.getUTCMinutes()).padStart(2, '0')
  const seconds = String(evaluatedAt.getUTCSeconds()).padStart(2, '0')
  evaluatedAt = `${year}-${month}-${day} ${hours}:${minutes}:${seconds}`
}
// 然后在 SQL 参数中使用 evaluatedAt（已经是 UTC 字符串）
```

**验收标准**：
- ✅ evaluated_at 正确存储为 UTC 时间
- ✅ 不依赖进程时区设置
- ✅ 前端显示时统一转换为北京时区（UTC+8）

#### 错误信息脱敏函数

/**
 * 错误信息脱敏（截断 + 移除敏感信息）
 * @param {string} errorMessage - 原始错误信息
 * @returns {string} - 脱敏后的错误信息（最多500字符）
 */
export function sanitizeErrorMessage(errorMessage) {
  if (!errorMessage) return null
  
  let sanitized = errorMessage
  
  // 移除敏感信息
  sanitized = sanitized.replace(/access_token[=:]\s*[\w-]+/gi, 'access_token=***')
  sanitized = sanitized.replace(/token[=:]\s*[\w-]+/gi, 'token=***')
  sanitized = sanitized.replace(/password[=:]\s*[\w-]+/gi, 'password=***')
  sanitized = sanitized.replace(/api_key[=:]\s*[\w-]+/gi, 'api_key=***')
  
  // 移除完整请求体（可能包含敏感信息）
  sanitized = sanitized.replace(/\{[\s\S]{200,}\}/g, '{...truncated...}')
  
  // 截断到500字符
  if (sanitized.length > 500) {
    sanitized = sanitized.substring(0, 497) + '...'
  }
  
  return sanitized
}
